### 2.1 quiz/index.jsx —— 计分逻辑

**现有逻辑（3维）**：
```js
const counts = next.reduce((acc, cur) => {
  acc[cur] = (acc[cur] || 0) + 1;
  return acc;
}, {});
const resultType = Object.entries(counts).sort((a, b) => b[1] - a[1])[0][0];
```

**新逻辑（4维）**：
```js
// 1. 计数 A/B/C/D
const counts = next.reduce((acc, cur) => {
  acc[cur] = (acc[cur] || 0) + 1;
  return acc;
}, { A: 0, B: 0, C: 0, D: 0 });

// 2. 排序找出最高维度
const sorted = Object.entries(counts).sort((a, b) => b[1] - a[1]);
const [top, second] = [sorted[0], sorted[1]];
const diff = top[1] - second[1];

// 3. 判断画像类型
let resultKey;
if (diff >= 3) {
  // 单维度主导（显著领先≥3）
  resultKey = top[0];  // "A" / "B" / "C" / "D"
} else {
  // 混合型：取前N个差值≤2的维度组合
  const mixed = sorted.filter(([k, v]) => top[1] - v <= 2);
  const dims = mixed.map(([k]) => k).sort().join('+');
  // dims 可能是 "A+B" / "A+B+C" / "A+B+C+D"
  // 需要从 results.json 里匹配对应的 key
  // 如果没有精确匹配，降级到最高单维度
  const possibleKeys = [dims, top[0]];
  resultKey = possibleKeys.find(k => resultsData.profiles[k]) || top[0];
}

// 4. 存储结果
Taro.setStorageSync('quizCounts', counts);
Taro.setStorageSync('quizResultKey', resultKey);
Taro.setStorageSync('quizAnswers', next);
```

### 2.2 result/index.jsx —— 画像匹配 + 展示

**画像匹配算法（从 counts 推导出 resultKey）**：
```js
function calculateProfile(counts) {
  // counts = { A: 8, B: 6, C: 5, D: 5 }
  const sorted = Object.entries(counts)
    .sort((a, b) => b[1] - a[1]); // 按次数降序
  
  const maxVal = sorted[0][1];
  const closeOnes = sorted.filter(([k, v]) => maxVal - v <= 2);
  
  if (closeOnes.length === 1) {
    // 单维度主导
    return closeOnes[0][0]; // "A"
  } else if (closeOnes.length === 2) {
    // 双维度混合
    return closeOnes.map(([k]) => k).sort().join('+'); // "A+B"
  } else if (closeOnes.length === 3) {
    // 三维度混合
    return closeOnes.map(([k]) => k).sort().join('+'); // "A+B+C"
  } else if (closeOnes.length === 4) {
    // 四维度平均
    // 检查是否严格相等
    const vals = sorted.map(([k, v]) => v);
    const isBalanced = vals.every(v => v === vals[0]);
    return isBalanced ? "A+B+C+D" : "balanced";
  }
}
```
