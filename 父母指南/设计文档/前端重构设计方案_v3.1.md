## 三、答题页改造方案

### 3.1 核心改造点

| 改造项 | 旧逻辑 | 新逻辑 |
|--------|--------|--------|
| 计分维度 | 3维（control/empathy/freedom） | 4维（A/B/C/D） |
| 题库数量 | 12道 | 按年龄段7道或全部28道 |
| 数据结构 | `options[].type: "control"` | `options[].type: "A"` |
| 进度计算 | `(index / 12) * 100` | `(index / totalQuestions) * 100` |
| 选项长度 | 15-25字短句 | 25-38字场景化描述 |
| 题目长度 | 15字短问句 | 50-80字场景描述 |

### 3.2 新计分逻辑

```js
// 1. 初始化计数器
const counts = { A: 0, B: 0, C: 0, D: 0 };

// 2. 用户选择时记录
const handleSelect = (type) => {
  // type = "A" / "B" / "C" / "D"
  const next = [...answers, type];
  counts[type]++;
  
  if (index + 1 < questions.length) {
    setAnswers(next);
    setIndex(index + 1);
  } else {
    // 3. 答题完成，计算画像
    const resultKey = calculateProfile(counts);
    Taro.setStorageSync('quizCounts', counts);
    Taro.setStorageSync('quizResultKey', resultKey);
    Taro.setStorageSync('quizAnswers', next);
    Taro.setStorageSync('quizQuestions', questions.map(q => q.id));
    Taro.redirectTo({ url: '/pages/result/index' });
  }
};

// 4. 画像匹配算法
function calculateProfile(counts) {
  const sorted = Object.entries(counts).sort((a, b) => b[1] - a[1]);
  const maxVal = sorted[0][1];
  const closeOnes = sorted.filter(([k, v]) => maxVal - v <= 2);
  
  if (closeOnes.length === 1) {
    return closeOnes[0][0];  // "A"
  } else if (closeOnes.length === 2) {
    return closeOnes.map(([k]) => k).sort().join('+');  // "A+B"
  } else if (closeOnes.length === 3) {
    return closeOnes.map(([k]) => k).sort().join('+');  // "A+B+C"
  } else {
    // 4个都接近，检查是否完全相等
    const vals = sorted.map(([k, v]) => v);
    const isBalanced = vals.every(v => v === vals[0]);
    return isBalanced ? "A+B+C+D" : "balanced";
  }
}
```

    ▼
